    \textbf{Uitleg gekoppeld aan de tabel:}

    \begin{itemize}
        \item \textbf{\color{olive}Movie} is het \textbf{relatieschema} $R$
        $R(\text{movid}, \text{title}, \text{year}, \text{rating})$

        \item De \textbf{\color{teal}kolomnamen} zijn de \textbf{attributen}
        $attr(R) = \{\text{movid}, \text{title}, \text{year}, \text{rating}\}$

        \item Het aantal attributen is 4, dus de \textbf{degree} is 4

        \item Elke rij in de tabel is één \textbf{tuple}
        Bijvoorbeeld:
        $<\text{tt0469494}, \text{There will be blood}, 2007, 8{,}1>$

        \item Elk attribuut heeft zijn \textbf{domein}:
        \begin{itemize}
            \item movid $\rightarrow$ string
            \item title $\rightarrow$ string
            \item year $\rightarrow$ integer
            \item rating $\rightarrow$ float
        \end{itemize}

        \item Alle rijen samen vormen de \textbf{relatie} $r(Movie)$
    \end{itemize}

    \subsection{Keys}
    \begin{itemize}
        \item \textbf{Key} is een attribuut of combinatie van attributen die een tuple uniek identificeert binnen een relatie.
        Bijvoorbeeld: movid is een key voor Movie.

        \item \textbf{Superkey} is een attribuut of combinatie van attributen die een tuple uniek identificeert, maar kan ook extra attributen bevatten.
        Bijvoorbeeld: (movid, title) is een superkey voor Movie.\\
        Definitie: Een set van attributen $K \subseteq attr(R)$ is een superkey als voor elke relatie $r(R)$: $\forall t_1, t_2 \in r(R): t_1[K] = t_2[K] \Rightarrow t_1 = t_2$.
        of: "Als twee tuples dezelfde waarden hebben voor alle attributen in K, dan zijn het dezelfde tuples."

        \item \textbf{Candidate key} is een minimale superkey, wat betekent dat het geen extra attributen bevat die niet nodig zijn voor unieke identificatie.
        Bijvoorbeeld: movid is een candidate key voor Movie.
        Definitie: Een set van attributen $K \subseteq attr(R)$ is een candidate key als K een superkey is en geen echte subset van K een superkey is.
        of: "Er is geen kleinere subset van attributen die ook een superkey is."

        \item \textbf{Primary key} is de gekozen candidate key die wordt gebruikt om tuples uniek te identificeren binnen een relatie.
        Bijvoorbeeld: movid is de primary key voor Movie.

        \item \textbf{Foreign key} is een attribuut of combinatie van attributen in een relatie die verwijst naar de primary key van een andere relatie.
        Bijvoorbeeld: in een relatie Actor met attribuut movid als foreign key die verwijst naar de primary key movid in Movie.
        Dit kan je gebruiken om aan te geven in welke films een acteur heeft gespeeld.
        Definitie: Een set attributen $K \in attr(R)$ is een foreign key van relatie $R_j$ als candidate key.
        $R_i[K]$ verwijst naar de primary key van relatie $R_j$.
        of: "De waarden van K in $R_i$ moeten overeenkomen met de waarden van de primary key in $R_j$."

        \item \textbf{Key constraints} zorgen ervoor dat de waarden van de key-attributen uniek zijn binnen de relatie en niet null mogen zijn.

        \item De \textbf{referential integrity constraints} zorgen ervoor dat de waarden van de foreign key-attributen overeenkomen met de waarden van de primary key in de gerelateerde relatie, of null zijn als er geen overeenkomst is.
    \end{itemize}

    \subsection{Selectie \& Projectie}

    \begin{itemize}
        \item \textbf{Selectie} ($\sigma_p$), $\sigma$ is een operator die rijen (tuples) selecteert uit een relatie op basis van het predicaat $p$.

        \item \textbf{and ($\land$)} en \textbf{or ($\lor$)} worden gebruikt om meerdere voorwaarden te combineren in het predicaat.
        Voorbeeld: $\sigma_{\text{year} > 2000 \land \text{rating} > 8}(Movie)$ selecteert alle films uit de Movie-relatie die zijn uitgebracht na het jaar 2000 en een beoordeling hoger dan 8 hebben.

        \item \textbf{Projectie} ($\pi_L$), $\pi$ is een operator die kolommen (attributen) selecteert uit een relatie op basis van de lijst van attributen $L$.
        Voorbeeld: $\pi_{\text{title}, \text{year}}(Movie)$ projecteert alleen de titel en het jaar van de films uit de Movie-relatie.

        \item \textbf{union} ($\cup$), de unie-operator combineert de resultaten van twee relaties door alle unieke tuples uit beide relaties te behouden.
        Voorbeeld: $R_1 \cup R_2$ geeft alle unieke tuples die voorkomen in zowel relatie $R_1$ als relatie $R_2$.

        \item \textbf{intersection} ($\cap$), de intersectie-operator retourneert de tuples die voorkomen in beide relaties.
        Voorbeeld: $R_1 \cap R_2$ geeft alle tuples die in zowel relatie $R_1$ als relatie $R_2$ voorkomen.

        \item \textbf{set difference} ($-$), de verschil-operator retourneert de tuples die voorkomen in de eerste relatie maar niet in de tweede.
        Voorbeeld: $R_1 - R_2$ geeft alle tuples die in relatie $R_1$ voorkomen maar niet in relatie $R_2$.

        \item \textbf{cartesian product} ($\times$), het cartesisch product combineert elke tuple van de eerste relatie met elke tuple van de tweede relatie.
        Voorbeeld: $R_1 \times R_2$ geeft alle mogelijke combinaties van tuples uit relatie $R_1$ en relatie $R_2$.

        \item \textbf{thetha-join} ($\bowtie_{\theta}$), de theta-join-operator combineert tuples uit twee relaties op basis van een opgegeven voorwaarde $\theta$.
        Voorbeeld: $R_1 \bowtie_{\text{R1.attr} = \text{R2.attr}} R_2$ combineert tuples uit relatie $R_1$ en relatie $R_2$ waar de waarden van het attribuut attr gelijk zijn.

        \item \textbf{natural join} ($\bowtie$), de natural join-operator combineert tuples uit twee relaties op basis van gemeenschappelijke attributen met dezelfde naam.
        Voorbeeld: $R_1 \bowtie R_2$ combineert tuples uit relatie $R_1$ en relatie $R_2$ waar de waarden van de gemeenschappelijke attributen gelijk zijn.

        \item \textbf{division} ($\div$), de division-operator retourneert de tuples uit de eerste relatie die overeenkomen met alle tuples in de tweede relatie.
        Voorbeeld: $R_1 \div R_2$ geeft alle tuples uit relatie $R_1$ die overeenkomen met alle tuples in relatie $R_2$ op basis van een gemeenschappelijk attribuut.

        \item \textbf{Assignment \& renaming} $T := $<alg_expr>, de assignment-operator wijst het resultaat van een relationele algebra-expressie toe aan een nieuwe relatie $T$.
        Voorbeeld: $T := \sigma_{\text{year} > 2000}(Movie)$ creëert een nieuwe relatie $T$ die alle films bevat die zijn uitgebracht na het jaar 2000.

        \item \textbf{Renaming on the fly} ($\rho(T)$(<alg_expr>), de renaming-operator wijst het resultaat van een relationele algebra-expressie toe aan een nieuwe relatie $T$ zonder een permanente naam te creëren.
        Voorbeeld: $\rho(T)(\sigma_{\text{year} > 2000}(Movie))$ creëert een tijdelijke relatie $T$ die alle films bevat die zijn uitgebracht na het jaar 2000, maar deze relatie is alleen beschikbaar binnen de context van deze expressie.
    \end{itemize}

    \section{Enity Relationship diagrams (ERD)}
    Als je een database wilt maken, dan is het belangrijk dat je het domein goed modelleert.
    Een ERD is een grafische weergave van de entiteiten, attributen en relaties binnen een domein.
    Het helpt bij het ontwerpen van databases door de structuur en de onderlinge verbanden duidelijk te maken.

    ER diagrammen hebben verschillende componenten:
    \begin{itemize}
        \item \textbf{Rechthoek} Geeft een entiteit weer, bijvoorbeeld een persoon, plaats of ding.
        \item \textbf{Ovaal} Geeft een attribuut weer, bijvoorbeeld een naam, leeftijd of adres.
        \item \textbf{Ruit} Geeft een relatie weer tussen entiteiten, bijvoorbeeld "werkt bij" of "heeft".
        \item \textbf{Lijn} Verbindt entiteiten, attributen en relaties om de structuur van het domein te tonen.
    \end{itemize}

    Deze lijnen/relaties kunnen ook kardinaliteiten hebben, zoals 1:1, 1:N of M:N, om aan te geven hoeveel entiteiten er betrokken zijn bij de relatie.\\
    1:1 (één-op-één) betekent dat één entiteit van het ene type overeenkomt met precies één entiteit van het andere type.\\
    1:N (één-op-veel) betekent dat één entiteit van het ene type overeenkomt met meerdere entiteiten van het andere type.\\
    M:N (veel-op-veel) betekent dat meerdere entiteiten van het ene type overeenkomen met meerdere entiteiten van het andere type.\\

    Ook kan je deelname aan een relatie aangeven als verplicht (1,\ldots) of optioneel (0,\ldots), wat aangeeft of een entiteit deel moet nemen aan de relatie of niet.

    \subsection{Functional dependencies}
    Stel je hebt een universum $R(A,B,C,D)$.
    Je hebt de regels $A \rightarrow B, B \rightarrow C, A \rightarrow C$

    Dan kan je concluderen dat $A \rightarrow AB \rightarrow ABC$, ofwel $A \rightarrow ABC$.\\\\

    Stel je hebt $A \rightarrow BCD$, dan kan je zeggen $A \rightarrow B$

    \begin{voorbeeld}{Voorbeeld van functionele afhankelijkheden}
        We hebben een schema $R(ABCDEFG)$ en een set FD's: $F = {C \rightarrow DE, A \rightarrow C, G \rightarrow D, B \rightarrow G}$\\
        We willen de closure van $A$ berekenen, oftewel $A^+$, om te zien welke attributen functioneel afhankelijk zijn van $A$.\\
        We beginnen met $A^+ = \{A\}$ (omdat $A$ altijd functioneel afhankelijk is van zichzelf).\\
        We kijken naar de FD's in $F$ en zien dat $A \rightarrow C$ in $F$ zit, dus we voegen $C$ toe aan $A^+$: $A^+ = \{A, C\}$.\\
        Nu kijken we opnieuw naar de FD's en zien dat $C \rightarrow DE$ in $F$ zit, en aangezien $C$ al in $A^+$ zit, kunnen we $D$ en $E$ ook toevoegen aan $A^+$: $A^+ = \{A, C, D, E\}$.\\
        We kijken nogmaals naar de FD's, maar er zijn geen FD's meer die we kunnen toepassen op de attributen in $A^+$, dus we zijn klaar.\\

        \\\\
        Stel wel willen $(AB)^+$ weten (notitie, dat is dus NIET $AB^+$, want dat is $A(B)^+$)\\

    \end{voorbeeld}

    \subsection{Decompositie}

    \begin{voorbeeld}{XYZ-regel (Verliesvrij)}
        Gegeven \(R(ABCDE)\) en FD's \(F = \{A \rightarrow C,\; C \rightarrow BE,\; A \rightarrow E\}\).\\
        Beschouw de decompositie \(R_1 = (AC)\) en \(R_2 = (ABDE)\).\\
        \vspace{\baselineskip}
        De gemeenschappelijke attributen zijn \(X = R_1 \cap R_2 = \{A\}\).\\ Schrijf \(Y = R_1 \setminus X = \{C\}\) en \(Z = R_2 \setminus X = \{B,D,E\}\).\\
        \vspace{\baselineskip}
        Volgens de XYZ-regel is de decompositie verliesvrij als \(X \rightarrow Y\) of \(X \rightarrow Z\) geldt (met betrekking tot \(F^+\)).\\
        Controleer \(A \rightarrow C\) en \(A \rightarrow BDE\):\\
        uit \(F\) volgt \(A \rightarrow C\) en \(A \rightarrow E\), maar er is geen \(A \rightarrow B\) en geen \(A \rightarrow D\).\\
        Dus geldt \(A \rightarrow C\) maar niet \(A \rightarrow BDE\).\\
        \vspace{\baselineskip}
        Conclusie: de decompositie \( (AC)\; (ABDE)\) is verliesvrij volgens de gegeven FD's.\\
    \end{voorbeeld}

    \vspace{\baselineskip}

    \begin{voorbeeld}{XYZ-regel (niet verliesvrij)}
        Gegeven \(R(ABCDE)\) en FD's \(F = \{A \rightarrow C,\; C \rightarrow BE,\; A \rightarrow E\}\).\\
        Beschouw de decompositie \(R_1 = (AC)\) en \(R_2 = (CBDE)\).\\
        \vspace{\baselineskip}
        De gemeenschappelijke attributen zijn \(X = R_1 \cap R_2 = \{C\}\).\\
        Schrijf \(Y = R_1 \setminus X = \{A\}\) en \(Z = R_2 \setminus X = \{B,D,E\}\).\\
        \vspace{\baselineskip}
        Volgens de XYZ-regel is de decompositie verliesvrij als \(X \rightarrow Y\) of \(X \rightarrow Z\) geldt (met betrekking tot \(F^+\)).\\
        Controleer \(C \rightarrow A\) en \(C \rightarrow BDE\):\\
        uit \(F\) volgt \(C \rightarrow B\) en \(C \rightarrow E\), maar er is geen \(C \rightarrow A\) en geen \(C \rightarrow D\).\\
        Dus geldt noch \(C \rightarrow A\) noch \(C \rightarrow BDE\).\\
        \vspace{\baselineskip}
        Conclusie: de decompositie \( (AC)\; (CBDE)\) is niet verliesvrij volgens de gegeven FD's.\\
    \end{voorbeeld}

    SQL (Structured Query Language) is een gestandaardiseerde programmeertaal die wordt gebruikt voor het beheren en manipuleren van relationele databases.
    SQL biedt een krachtige en flexibele manier om gegevens op te vragen, in te voegen, bij te werken en te verwijderen, evenals om de structuur van databases te definiëren en te beheren.

    Om SQL uit te leggen gebruik ik deze tabel als voorbeeld:

    \vspace{\baselineskip}

    \noindent\textbf{CREATE TABLE} wordt gebruikt om een nieuwe tabel in de database te maken.

    \begin{lstlisting}[style=SQL]
CREATE TABLE Books ( -- de naam van de tabel
    book_id INT PRIMARY KEY, -- een integer kolom die dient als primaire sleutel
    title VARCHAR(255) NOT NULL, -- string, maximaal 255 tekens, mag niet null zijn
    author VARCHAR(255) NOT NULL,
    year INT,
    genre VARCHAR(100)
);
    \end{lstlisting}

    \subsection{Indexen}
    Indexen zijn datastructuren die worden gebruikt om de snelheid van gegevensopvragingen in een database te verbeteren.
    Ze werken als een soort inhoudsopgave voor de database, waardoor het sneller is om specifieke records te vinden zonder de hele tabel te hoeven doorzoeken.
    Indexen kunnen worden gemaakt op één of meer kolommen van een tabel, en ze kunnen verschillende vormen aannemen, zoals B-tree indexen, hash indexen, of bitmap indexen, afhankelijk van het type gegevens en de query's die worden uitgevoerd.
    Het gebruik van indexen kan de prestaties van zoekopdrachten aanzienlijk verbeteren, vooral bij grote datasets, maar het kan ook extra opslagruimte vereisen en de prestaties van schrijfoperaties (zoals INSERT, UPDATE, DELETE) verminderen, omdat de indexen ook moeten worden bijgewerkt.

    \subsection{Boyce-Codd Normal Form (BCNF)}
    De \textbf{Boyce-Codd Normal Form (BCNF)} is een strengere vorm van normalisatie dan de derde normaalvorm (3NF).
    Een relatie is in BCNF als voor elke functionele afhankelijkheid \(X \rightarrow Y\) geldt dat \(X\) een superkey is.
    Dit betekent dat er geen functionele afhankelijkheden mogen zijn waarbij de determinant \(X\) geen superkey is.
    BCNF elimineert dus alle anomalieën die kunnen ontstaan door functionele afhankelijkheden waarbij de determinant geen superkey is, waardoor het ontwerp van de database robuuster en efficiënter wordt.

    \begin{voorbeeld}{BCNF voorbeeld}
        \begin{SimpleTable}[s{}s{}s{}]{}{}
            \TableHeader{Student & Teacher & Course}
            \TableRow{Alice & Dr. Smith & Math}
            \TableRow{Alice & Dr. Jones & History}
            \TableRow{Bob & Dr. Smith & Math}
            \TableRow{Bob & Dr. Stoned & History}
        \end{SimpleTable}

        \vmed

        Hieruit kunnen we opmaken dat de FD's zijn: \\
        \begin{itemize}
            \item (Student, Course) $\rightarrow$ Teacher (omdat elke student in een bepaald vak maar één docent heeft)
            \item (Student, Teacher) $\rightarrow$ Course (omdat elke student bij een bepaalde docent maar één vak volgt)
            \item Teacher $\rightarrow$ Course (omdat elke docent maar één vak geeft)
        \end{itemize}

        Hierdoor zijn de candidate keys (Student, Course) en (Student, Teacher), maar Teacher is geen superkey omdat er meerdere docenten zijn die hetzelfde vak geven.
        Dus deze relatie is niet in BCNF.
        We kunnen deze relatie decomponeren in twee relaties: \\

        \begin{itemize}
            \item R1(Teacher, Course) met de FD Teacher $\rightarrow$ Course
            \item R2(Student, Teacher) met de FD (Student, Teacher) $\rightarrow$ Course
        \end{itemize}

        Nu zijn beide relaties in BCNF, omdat in R1 Teacher een superkey is en in R2 (Student, Teacher) een superkey is.
    \end{voorbeeld}

    \subsection{Minimal cover calculation}
    Een minimal cover van een set functionele afhankelijkheden is een equivalentie van die set waarbij alle functionele afhankelijkheden in de minimal cover minimaal zijn, wat betekent dat er geen overbodige functionele afhankelijkheden zijn en dat de rechterkant van elke functionele afhankelijkheid slechts één attribuut bevat.
    Het berekenen van een minimal cover gaat als volgt:

    \noindent Stel je hebt de volgende set functionele afhankelijkheden:\\
    \[F = \{A \rightarrow D,B \rightarrow HA, GH \rightarrow AKL, CH \rightarrow BK, G \rightarrow BC\}\]
    Calculate a minimal cover for \textbf{F}

    \begin{multicols}{2}
        \item Stap 1 is om de rechterkant van elke functionele afhankelijkheid te splitsen in enkelvoudige attributen:

        \newcolumn

        \begin{itemize}
            \item A $\rightarrow$ D
            \item B $\rightarrow$ H
            \item B $\rightarrow$ A
            \item GH $\rightarrow$ A
            \item GH $\rightarrow$ K
            \item GH $\rightarrow$ L
            \item CH $\rightarrow$ B
            \item CH $\rightarrow$ K
            \item G $\rightarrow$ B
            \item G $\rightarrow$ C
        \end{itemize}
    \end{multicols}

    \begin{multicols}{2}
        \item Stap 2 is de linkerkant van elke functionele afhankelijkheid minimaliseren door te controleren of er overbodige attributen zijn:

        \newcolumn

        \begin{itemize}
            \item A $\rightarrow$ D
            \item B $\rightarrow$ H
            \item B $\rightarrow$ A
            \item G\textcolor{kindred}{H} $\rightarrow$ A ($G^+ = GBCHAKLD$, dus G $\rightarrow$ A)
            \item G\textcolor{kindred}{K} $\rightarrow$ A ($G^+ = GBCHAKLD$, dus G $\rightarrow$ K)
            \item G\textcolor{kindred}{L} $\rightarrow$ A ($G^+ = GBCHAKLD$, dus G $\rightarrow$ L)
            \item CH $\rightarrow$ B (Kan niet versimpeld worden, want $C^+ \rightarrow {C}$ en $H^+ \rightarrow {H}$)
            \item CH $\rightarrow$ K (Kan niet versimpeld worden, want $C^+ \rightarrow {C}$ en $H^+ \rightarrow {H}$)
            \item G $\rightarrow$ B
            \item G $\rightarrow$ C
        \end{itemize}
    \end{multicols}

    \begin{multicols}{2}
        \item Stap 3 is om te kijken over er overbodige functionele afhankelijkheden zijn door te controleren of ze kunnen worden afgeleid uit de andere functionele afhankelijkheden:

        \newcolumn

        \begin{itemize}
            \item A $\rightarrow$ D
            \item B $\rightarrow$ H
            \item B $\rightarrow$ A
            \item \textcolor{kindred}{G $\rightarrow$ A} (kan worden afgeleid uit G $\rightarrow$ B en B $\rightarrow$ A, dus overbodig)
            \item \textcolor{kindred}{G $\rightarrow$ K} (kan worden afgeleid uit G $\rightarrow$ C en C $\rightarrow$ K, dus overbodig)
            \item G $\rightarrow$ L (kan \textit{niet meer} worden afgeleid, want G $\rightarrow$ A is al weg)
            \item CH $\rightarrow$ B
            \item CH $\rightarrow$ K
            \item G $\rightarrow$ B
            \item G $\rightarrow$ C
        \end{itemize}
    \end{multicols}

    De minimal cover van F is dus:
    \[F_{min} = \{A \rightarrow D, B \rightarrow HA, G \rightarrow LBC, CH \rightarrow BK\}\]
    (opmerking: G $\rightarrow$ LBC is een notatie die betekent dat G $\rightarrow$ L, G $\rightarrow$ B en G $\rightarrow$ C allemaal in de minimal cover zitten)
