    \textbf{Uitleg gekoppeld aan de tabel:}

    \begin{itemize}
        \item \textbf{\color{olive}Movie} is het \textbf{relatieschema} $R$
        $R(\text{movid}, \text{title}, \text{year}, \text{rating})$

        \item De \textbf{\color{teal}kolomnamen} zijn de \textbf{attributen}
        $attr(R) = \{\text{movid}, \text{title}, \text{year}, \text{rating}\}$

        \item Het aantal attributen is 4, dus de \textbf{degree} is 4

        \item Elke rij in de tabel is één \textbf{tuple}
        Bijvoorbeeld:
        $<\text{tt0469494}, \text{There will be blood}, 2007, 8{,}1>$

        \item Elk attribuut heeft zijn \textbf{domein}:
        \begin{itemize}
            \item movid $\rightarrow$ string
            \item title $\rightarrow$ string
            \item year $\rightarrow$ integer
            \item rating $\rightarrow$ float
        \end{itemize}

        \item Alle rijen samen vormen de \textbf{relatie} $r(Movie)$
    \end{itemize}

    \subsection{Keys}
    \begin{itemize}
        \item \textbf{Key} is een attribuut of combinatie van attributen die een tuple uniek identificeert binnen een relatie.
        Bijvoorbeeld: movid is een key voor Movie.

        \item \textbf{Superkey} is een attribuut of combinatie van attributen die een tuple uniek identificeert, maar kan ook extra attributen bevatten.
        Bijvoorbeeld: (movid, title) is een superkey voor Movie.\\
        Definitie: Een set van attributen $K \subseteq attr(R)$ is een superkey als voor elke relatie $r(R)$: $\forall t_1, t_2 \in r(R): t_1[K] = t_2[K] \Rightarrow t_1 = t_2$.
        of: "Als twee tuples dezelfde waarden hebben voor alle attributen in K, dan zijn het dezelfde tuples."

        \item \textbf{Candidate key} is een minimale superkey, wat betekent dat het geen extra attributen bevat die niet nodig zijn voor unieke identificatie.
        Bijvoorbeeld: movid is een candidate key voor Movie.
        Definitie: Een set van attributen $K \subseteq attr(R)$ is een candidate key als K een superkey is en geen echte subset van K een superkey is.
        of: "Er is geen kleinere subset van attributen die ook een superkey is."

        \item \textbf{Primary key} is de gekozen candidate key die wordt gebruikt om tuples uniek te identificeren binnen een relatie.
        Bijvoorbeeld: movid is de primary key voor Movie.

        \item \textbf{Foreign key} is een attribuut of combinatie van attributen in een relatie die verwijst naar de primary key van een andere relatie.
        Bijvoorbeeld: in een relatie Actor met attribuut movid als foreign key die verwijst naar de primary key movid in Movie.
        Dit kan je gebruiken om aan te geven in welke films een acteur heeft gespeeld.
        Definitie: Een set attributen $K \in attr(R)$ is een foreign key van relatie $R_j$ als candidate key.
        $R_i[K]$ verwijst naar de primary key van relatie $R_j$.
        of: "De waarden van K in $R_i$ moeten overeenkomen met de waarden van de primary key in $R_j$."

        \item \textbf{Key constraints} zorgen ervoor dat de waarden van de key-attributen uniek zijn binnen de relatie en niet null mogen zijn.

        \item De \textbf{referential integrity constraints} zorgen ervoor dat de waarden van de foreign key-attributen overeenkomen met de waarden van de primary key in de gerelateerde relatie, of null zijn als er geen overeenkomst is.
    \end{itemize}

    \subsection{Selectie \& Projectie}

    \begin{itemize}
        \item \textbf{Selectie} ($\sigma_p$), $\sigma$ is een operator die rijen (tuples) selecteert uit een relatie op basis van het predicaat $p$.

        \item \textbf{and ($\land$)} en \textbf{or ($\lor$)} worden gebruikt om meerdere voorwaarden te combineren in het predicaat.
        Voorbeeld: $\sigma_{\text{year} > 2000 \land \text{rating} > 8}(Movie)$ selecteert alle films uit de Movie-relatie die zijn uitgebracht na het jaar 2000 en een beoordeling hoger dan 8 hebben.

        \item \textbf{Projectie} ($\pi_L$), $\pi$ is een operator die kolommen (attributen) selecteert uit een relatie op basis van de lijst van attributen $L$.
        Voorbeeld: $\pi_{\text{title}, \text{year}}(Movie)$ projecteert alleen de titel en het jaar van de films uit de Movie-relatie.

        \item \textbf{union} ($\cup$), de unie-operator combineert de resultaten van twee relaties door alle unieke tuples uit beide relaties te behouden.
        Voorbeeld: $R_1 \cup R_2$ geeft alle unieke tuples die voorkomen in zowel relatie $R_1$ als relatie $R_2$.
        
        \item \textbf{intersection} ($\cap$), de intersectie-operator retourneert de tuples die voorkomen in beide relaties.
        Voorbeeld: $R_1 \cap R_2$ geeft alle tuples die in zowel relatie $R_1$ als relatie $R_2$ voorkomen.

        \item \textbf{set difference} ($-$), de verschil-operator retourneert de tuples die voorkomen in de eerste relatie maar niet in de tweede.
        Voorbeeld: $R_1 - R_2$ geeft alle tuples die in relatie $R_1$ voorkomen maar niet in relatie $R_2$.

        \item \textbf{cartesian product} ($\times$), het cartesisch product combineert elke tuple van de eerste relatie met elke tuple van de tweede relatie.
        Voorbeeld: $R_1 \times R_2$ geeft alle mogelijke combinaties van tuples uit relatie $R_1$ en relatie $R_2$.

        \item \textbf{thetha-join} ($\bowtie_{\theta}$), de theta-join-operator combineert tuples uit twee relaties op basis van een opgegeven voorwaarde $\theta$.
        Voorbeeld: $R_1 \bowtie_{\text{R1.attr} = \text{R2.attr}} R_2$ combineert tuples uit relatie $R_1$ en relatie $R_2$ waar de waarden van het attribuut attr gelijk zijn.

        \item \textbf{natural join} ($\bowtie$), de natural join-operator combineert tuples uit twee relaties op basis van gemeenschappelijke attributen met dezelfde naam.
        Voorbeeld: $R_1 \bowtie R_2$ combineert tuples uit relatie $R_1$ en relatie $R_2$ waar de waarden van de gemeenschappelijke attributen gelijk zijn.

        \item \textbf{division} ($\div$), de division-operator retourneert de tuples uit de eerste relatie die overeenkomen met alle tuples in de tweede relatie.
        Voorbeeld: $R_1 \div R_2$ geeft alle tuples uit relatie $R_1$ die overeenkomen met alle tuples in relatie $R_2$ op basis van een gemeenschappelijk attribuut.
    \end{itemize}

    \begin{voorbeeld}{bibliotheek schema}
        Stel je hebt de volgende schema's voor een bibliotheekdatabase:
        \begin{itemize}
            \item \textbf{Book}(\text{book\_id}, \text{title}, \text{author}, \text{year})
            \item \textbf{Member}(\text{member\_id}, \text{name}, \text{join\_date})
            \item \textbf{Loan}(\text{loan\_id}, \text{book\_id}, \text{member\_id}, \text{loan\_date}, \text{return\_date})
        \end{itemize}

        \vsmall

        \textbf{Geef alle namen van lezers die minstens één boek hebben geleend van Dickens.}\\
        \[\pi_{\text{name}}(\text{Reader}\bowtie (\text{Loan}\bowtie \sigma_{\text{author} = 'Dickens'}(\text{Book})))\]

        Uitleg:\\
        \begin{itemize}
            \item $\sigma_{\text{author} = 'Dickens'}(\text{Book})$ selecteert alle boeken geschreven door Dickens.
            \item $\text{Loan}\bowtie \sigma_{\text{author} = 'Dickens'}(\text{Book})$ combineert de leningen met de geselecteerde boeken om alle leningen van Dickens' boeken te vinden.
            \item $\text{Reader}\bowtie (\text{Loan}\bowtie \sigma_{\text{author} = 'Dickens'}(\text{Book}))$ combineert de lezers met de leningen van Dickens' boeken om alle lezers te vinden die deze boeken hebben geleend.
            \item $\pi_{\text{name}}(\ldots)$ projecteert alleen de namen van deze lezers.
        \end{itemize}


    \end{voorbeeld}
