\makeatletter
\newcount\TOClinecount

\newcommand{\partialtableofcontents}[1]{%
    \begingroup
    \TOClinecount=0
    \let\origcontentsline\contentsline
    \def\contentsline##1##2##3##4{%
        \advance\TOClinecount by 1
        \ifnum\TOClinecount>#1\relax
        \expandafter\@gobblefour
        \else
            \origcontentsline{##1}{##2}{##3}{##4}%
        \fi
    }%
    \@input{\jobname.toc}%
    \endgroup
}

\newcommand{\remainingtableofcontents}[1]{%
    \begingroup
    \TOClinecount=0
    \let\origcontentsline\contentsline
    \def\contentsline##1##2##3##4{%
        \advance\TOClinecount by 1
        \ifnum\TOClinecount>#1\relax
        \origcontentsline{##1}{##2}{##3}{##4}%
        \fi
    }%
    \@starttoc{toc}%
    \endgroup
}

\newcommand{\TOCmoreThan}[1]{%
    \begingroup
    \TOClinecount=0%
    \let\origcontentsline\contentsline
    \def\contentsline##1##2##3##4{%
        \advance\TOClinecount by 1\relax
    }%
    \@input{\jobname.toc}%
    \ifnum\TOClinecount>#1\relax
        \endgroup
        \null\newpage
    \else
        \endgroup
    \fi
}
